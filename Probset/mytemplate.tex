\usepackage{graphicx}
\usepackage[most]{tcolorbox}
\usepackage{xcolor}

\definecolor{myg}{RGB}{56, 140, 70}
\definecolor{myb}{RGB}{45, 111, 177}
\definecolor{myexamplebg}{RGB}{238, 238, 238}
\definecolor{myexampleframe}{RGB}{118, 171, 174}
\definecolor{myexampletitle}{RGB}{42, 127, 127}

% Problem box.
\newtcbtheorem{problem}{Problem}{
    enhanced, 
    breakable, 
    colback = white, 
    colframe = myb!80!black,
    attach boxed title to top left = {yshift*=-\tcboxedtitleheight},
    fonttitle = \bfseries,
    title = {#2},
    boxed title size = title,
    boxed title style = {
        sharp corners,
	rounded corners = northwest,
	colback = tcbcolframe,
	boxrule = 0pt,
    },
    underlay boxed title = {
        \path[fill=tcbcolframe] (title.south west) -- (title.south east) to[out=0, in=180]([xshift=5mm]title.east) -- (title.center-|frame.east)[] |- (frame.north) -| cycle;
    },
    #1
}{def}

% Solution box.
\newtcbtheorem{solution}{Solution}{
    enhanced, 
    breakable,
    colback = myg!10,
    frame hidden,
    boxrule = 0sp,
    borderline west = {2pt}{0pt}{myg},
    sharp corners,
    detach title,
    before upper = \tcbtitle\par\smallskip, 
    coltitle = myg!85!black,
    fonttitle = \bfseries\sffamily,
    description font = \mdseries,
    separator sign none,
    segmentation style = {solid, myg}
}{th}

% Example box.
\newtcbtheorem{example}{Example}{
    colback=myexamplebg, 
    breakable, 
    colframe=myexampleframe,
    coltitle=myexampletitle,
    boxrule=1pt,
    sharp corners,
    detach title, 
    before upper=\tcbtitle\par\smallskip,
    fonttitle=\bfseries,
    description font=\mdseries,
    separator sign none,
    description delimiters parenthesis
}{ex}
